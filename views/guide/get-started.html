# 引入 @sscui

你可以引入整个 @sscui，或是根据需要仅引入部分组件。我们先介绍如何引入完整的 @sscui。

# 完整引入

在 main.js 中写入以下内容：

import Vue from 'vue';
import ClassroomUi from '@sscui/classroom-ui';
import App from './App.vue';

Vue.use(ClassroomUi);

new Vue({
  el: '#app',
  render: h => h(App)
});

# 按需引入

如果你只希望引入部分组件，比如 Button 和 svgIcon，那么需要在 main.js 中写入以下内容：

import Vue from 'vue';
import SvgIcon from '@sscui/svg-icon';
import Button from '@sscui/button';
import App from './App.vue';

Vue.component(Button.name, Button);
Vue.component(SvgIcon.name, SvgIcon);
/* 或写为
 * Vue.use(SvgIcon)
 * Vue.use(Button)
 */

new Vue({
  el: '#app',
  render: h => h(App)
});

javascript

export default {
  props: {
    // Clock 模块
    startTime: { // 开课时间（时间戳）
      type: Number,
      default: 0
    },
    time: { // 这节课时长(分钟)
      type: Number,
      default: 25
    },
    manualDuration: { // 上课多久后可手动操作(分钟)
      type: Number,
      default: 25
    },
    maxDuration: { // 这节课的最长时间(分钟)
      type: Number,
      default: 40
    },
    // Signal 模块
    networkRtt: { // 网络 rtt 50以内为好，100左右一般 200以上肯定是差了 50 100 150 200
      type: String,
      default: '0ms'
    },
    // Settings 模块
    configSettingsOptions: {
      type: Object,
      default: () => ({
        microphone: {
          id: '',
          volume: 50
        },
        camera: {
          id: '',
          status: false
        },
        speaker: {
          id: ''
        }
      })
    },
    use: {
      type: String,
      default: 'navigator',
      validator: val => ['navigator', 'mediaSDK'].indexOf(val) > -1
    },
    // Flushed 模块
    offlineStatus: {
      type: Boolean,
      default: false
    },
    activeingStatus: {
      type: Boolean,
      default: false
    },
    answeringStatus: {
      type: Boolean,
      default: false
    },
    // Operations 模块
    operationsStatus: {
      type: String,
      default: '',
      validator: val => ['startMeetuping', 'quizContesting', 'groupDiscussioning', 'thumbsUping', 'matchQuestion', 'voiceQuestion', ''].indexOf(val) > -1
    },
    // CoursewarePanel 模块
    remainingSeconds: {
      type: Number,
      default: 0
    },
    voiceSeconds: {
      type: Number,
      default: 10
    },
    questionResultData: {
      type: Object,
      default: () => ({})
    },
    voiceConfigData: {
      type: Object,
      default: () => ({})
    },
    uid: {
      type: Number,
      default: 0
    },
    barrageList: {
      type: Array,
      default () {
        return []
      }
    },
    // TeacherPanel 模块
    userVideoStreamStatus: {
      type: Object,
      default: () => {
      }
    },
    gender: {
      type: String,
      default: 'FEMALE',
      validator: val => ['FEMALE', 'MALE'].indexOf(val) > -1
    },
    teacherId: {
      type: Number,
      default: 0
      // required: true,
    },
    cameraId: String,
    onlineStatus: {
      type: Boolean,
      default: true
    },
    cameraStatus: {
      type: Boolean,
      default: false
    },
    voiceStatus: {
      type: Boolean,
      default: false
    },
    screenStatus: {
      type: Boolean,
      default: false
    },
    giftNums: {
      type: Number,
      default: 0
    },
    // StudentGroupPanel 模块
    groupInfo: {
      type: Object,
      default: () => ({})
    },
    // Panels 模块
    tips: {
      type: String,
      default: ''
    },
    notice: {
      type: String,
      default: ''
    },
    // Leaderboard 模块
    studentNum: {
      type: Number,
      default: 0
    },
    rankList: {
      type: Array,
      default: () => ([])
    }
  },
  data () {
    return {
      activeStatus: this.operationsStatus
    }
  },
  methods: {
    // Clock 回调
    endClassCb () {
      this.$emit('endClassCb')
    },
    // Settings 回调
    handleSettingsChange (data) {
      this.$emit('handleSettingsChange', data)
    },
    handleSettingsPlayStream (data) {
      this.$emit('handleSettingsPlayStream', data)
    },
    // Flushed 回调
    closeFlushedTipsCb (status) {
      this.$emit('closeFlushedTipsCb', status)
    },
    handleRefresh () {
      this.$emit('refresh')
    },
    // Operations 回调
    handleOperationsStatus (result) {
      this.$emit('handleOperationsStatus', result)
    },
    // CoursewarePanel 回调
    handleEndFeedbackStatus () {
      this.activeStatus = ''
    },
    requestVoiceAnswerData (questionId) {
      this.$emit('requestVoiceAnswerData', questionId)
    },
    changeBarrageMsgStatus (status) {
      this.$emit('changeBarrageMsgStatus', status)
    },
    changeBarrageScrollStatus (status) {
      this.$emit('changeBarrageScrollStatus', status)
    },
    changePlayingStudentId (studentId) {
      this.$emit('changePlayingStudentId', studentId)
    },
    closeVoiceFeedback () {
      this.$emit('closeVoiceFeedback')
    },
    // TeacherPanel  回调
    toggleCamera (cameraStatus, { uid, target }) {
      this.$emit('toggleCamera', cameraStatus, {
        uid,
        target
      })
    },
    toggleVoice (voiceStatus) {
      this.$emit('toggleVoice', voiceStatus)
    },
    toggleScreen (screenStatus) {
      this.$emit('toggleScreen', screenStatus)
    },
    // StudentGroupPanel 回调
    encourage ({ msgtype, groupId }) {
      this.$emit('encourage', {
        msgtype,
        groupId
      })
    },
    toggleStudentCamera (cameraStatus, screen, groupId) {
      this.$emit('toggleStudentCamera', cameraStatus, screen, groupId)
    },
    timeoutPullGroupInfo (cb) {
      this.$emit('timeoutPullGroupInfo', cb)
    },
    pullGroupInfo (cb) {
      this.$emit('pullGroupInfo', cb)
    },
    // Panels  回调
    publishNotice (context) {
      this.$emit('publishNotice', context)
    },
    // Leaderboard 回调
    handleRankEncourage (groupIds) {
      this.$emit('handleRankEncourage', groupIds)
    }
  },
  watch: {
    operationsStatus (status) {
      if (this.activeStatus !== status) {
        this.activeStatus = status
      }
    },
    activeStatus (status) {
      this.$emit('changeOperationsStatus', status)
    }
  }
}
